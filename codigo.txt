# Inicio del programa
mostrar("Iniciando ejecución...");

# Declaración y asignación de variables
numero1 = 10;
numero2 = 5;
resultado = 0;

# Operaciones aritméticas
resultado = numero1 + numero2;
mostrar("Suma:", resultado);

resultado = numero1 - numero2;
mostrar("Resta:", resultado);

resultado = numero1 * numero2;
mostrar("Multiplicación:", resultado);

resultado = numero1 / numero2;
mostrar("División:", resultado);

resultado = numero1 % numero2;
mostrar("Módulo:", resultado);

# Condicional si-sino
si (numero1 > numero2) {
    mostrar("numero1 es mayor que numero2");
} sino {
    mostrar("numero1 no es mayor que numero2");
}

# Bucle mientras
contador = 1;
mientras (contador <= 3) {
    mostrar("Iteración mientras:", contador);
    contador = contador + 1;
}

# Bucle para
para (i = 1; i <= 5; i = i + 1) {
    mostrar("Iteración para:", i);
}

mostrar("Ejecución finalizada.");
